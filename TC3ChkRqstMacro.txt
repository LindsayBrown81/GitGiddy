Sub TC3ChkRqstMacro()

' In the check request log, click anywhere on the tab you want to log to. Take note of line # you want to log to.
' On the check request, click anywhere on the active tab.
' In Visual Basic for Applications (VBA), hit Ctrl + F and type in the row/range number that the macro last ran on
' (see Range("F---").Select).
' Click the Replace button. Type in the row/range number for the row in which you want to log the check request info.
' Click the Replace All button. 10 replacements made (as of 11/09/15). Click Okay. Close that dialogue box or
'  move it to the side.
' Run the macro. Make sure it worked before you save either the check request or the log.

    Range("C6").Select 'Select Payable to value from check request
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("G34").Select '*Whatever number is listed in this range, Find and Replace it/Paste into vendor column on check request log
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("M12").Select 'Select dollar amount value from check request
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("F34").Select 'Paste into Amount column on check request log
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    ActiveWindow.SmallScroll Down:=6
    Range("C32").Select 'Select Date value from check request
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("D34").Select 'Paste into Date column on check request log
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("K35").Select 'Select Invoice# value from check request
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("E34").Select 'Paste into Invoice# column on check request log
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("E37").Select 'Select Invoice Date value from check request
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("D34").Select 'Paste into Invoice Date column on check request log
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("G37").Select 'Select Description from check request
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("H34").Select 'Paste into Description column on check request log
    ActiveSheet.Paste
    
    Range("C34").Select 'Staying on the active tab, select the FY column on the row we designated with find and replace
    ActiveCell.Offset(-1, 0).Select 'Move up one row, over none.
    ActiveCell.Copy 'Copy that FY value
    ActiveCell.Offset(1, 0).Select 'Move back down one row, over none.
    ActiveSheet.Paste 'Paste that FY value
    
    'Remove borders from row on check request log
    Range("C34", Range("C34").End(xlToRight)).Select 'Select from the FY column to the end of the row
    Selection.Borders(xlDiagonalDown).LineStyle = xlNone 'No borders of any xl style on this row
    Selection.Borders(xlDiagonalUp).LineStyle = xlNone
    Selection.Borders(xlEdgeLeft).LineStyle = xlNone
    Selection.Borders(xlEdgeTop).LineStyle = xlNone
    Selection.Borders(xlEdgeBottom).LineStyle = xlNone
    Selection.Borders(xlEdgeRight).LineStyle = xlNone
    Selection.Borders(xlInsideVertical).LineStyle = xlNone
    Selection.Borders(xlInsideHorizontal).LineStyle = xlNone
    
    'Left align and bottom align columns FY through Invoice Date
    Range("C:F").Select 'Select columns FY through Invoice Date
    With Selection
        .HorizontalAlignment = xlRight
        .VerticalAlignment = xlBottom
        .WrapText = False
        .Orientation = 0
        .AddIndent = False
        .IndentLevel = 0
        .ShrinkToFit = False
        .ReadingOrder = xlContext
        .MergeCells = False
    
    'Format the Amount column as currency
    End With
    Range("F34").Select 'Select the Amount column
    Selection.Style = "Currency"
    ActiveWindow.ActivatePrevious
    ActiveWindow.ActivatePrevious

End Sub
