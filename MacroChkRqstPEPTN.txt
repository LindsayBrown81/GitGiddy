Sub ChkRqstToNTPEPLog()
'
' ChkRqstToNTPEPLog Macro
' In the check request log, click anywhere on the tab you want to log to.
' On the check request, click anywhere on the active tab.
' In Visual Basic for Applications (VBA), hit Ctrl + F and type in the row/range number that the macro last ran on (see Range("F---").Select).
' Click the Replace button. Type in the row/range number for the row in which you want to log the check request info.
' Click the Replace All button. 10 replacements made (as of 10/07/15). Click Okay. Close that dialogue box or move it to the side.
' Run the macro. Make sure it worked before you save either the check request or the log.

    Range("C6").Select
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("F129").Select
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("M12").Select
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("E129").Select
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    ActiveWindow.SmallScroll Down:=6
    Range("C32").Select
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("B129").Select
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("K35").Select
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("C129").Select
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("E37").Select
    Application.CutCopyMode = False
    Selection.Copy
    ActiveWindow.ActivatePrevious
    Range("D129").Select
    ActiveSheet.Paste
    ActiveWindow.ActivatePrevious
    Range("G37").Select
    Application.CutCopyMode = False
    Selection.Copy
    
    
    ActiveWindow.ActivatePrevious
    Range("G129").Select
    ActiveSheet.Paste
    Range("A129").Select
    ActiveCell.Offset(-1, 0).Select
    ActiveCell.Copy
    ActiveCell.Offset(1, 0).Select
    ActiveSheet.Paste
    
    Range("A129", Range("A129").End(xlToRight)).Select
    Selection.Borders(xlDiagonalDown).LineStyle = xlNone
    Selection.Borders(xlDiagonalUp).LineStyle = xlNone
    Selection.Borders(xlEdgeLeft).LineStyle = xlNone
    Selection.Borders(xlEdgeTop).LineStyle = xlNone
    Selection.Borders(xlEdgeBottom).LineStyle = xlNone
    Selection.Borders(xlEdgeRight).LineStyle = xlNone
    Selection.Borders(xlInsideVertical).LineStyle = xlNone
    Selection.Borders(xlInsideHorizontal).LineStyle = xlNone
    
    Range("C:D").Select
    With Selection
        .HorizontalAlignment = xlRight
        .VerticalAlignment = xlBottom
        .WrapText = False
        .Orientation = 0
        .AddIndent = False
        .IndentLevel = 0
        .ShrinkToFit = False
        .ReadingOrder = xlContext
        .MergeCells = False
    
    End With
    Range("E129").Select
    Selection.Style = "Currency"
    ActiveWindow.ActivatePrevious
    ActiveWindow.ActivatePrevious
End Sub
